#include<iostream>
using namespace std;

int main()
{
    int i;
    float f;
    char ch;
    string str;

    cout<<"Enter integer input: ";
    cin>>i;
    cout<<"Enter float input: ";
    cin>>f;
    cout<<"Enter char input: ";
    cin>>ch;
    cout<<"Enter String input: ";
    cin>>str;

    cout<<"integer output: "<<i<<endl;
    cout<<"float output: "<<f<<endl;
    cout<<"Character output: "<<ch<<endl;
    cout<<"String output: "<<str<<endl;

    return 0;
}
